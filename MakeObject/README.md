# Object Literal

```
let Student = {
  name: "Abdul Aziz",
  age: 22,
  hoby: function (hobbies) {
    console.log(`my hobi ${hobbies}`);
  },
};
let Student2 = {
  name: "Azizah",
  age: 20,
  hoby: function (hobbies) {
    console.log(`my hobi ${hobbies}`);
  },
};

console.log(Student.hoby("coding"));
console.log(Student2.hoby("design"));

```
